# 1. Connection
# Syntax: /connect <hostname> <port> <password> <nickname>
irssi -c localhost -p 6667 -w pwd123 -n user
OR
/connect localhost 6667 pwd123 user

# 2. Channel Operations
# Syntax: /join <channel>
/join #test

# Syntax: /part <channel>
/part #test

# 3. Messaging
# Syntax: /msg <nickname> <message>
/msg user Hello there!

# 4. Topic
# Syntax: /topic <channel> <topic>
/topic #test Welcome to the test channel

# 5. Modes
# Syntax: /mode <channel> <mode> [args]

# Invite-only (+i)
/mode #test +i
/mode #test -i

# Topic restriction (+t)
/mode #test +t
/mode #test -t

# Key/Password (+k)
/mode #test +k secret123
/mode #test -k secret123

# Operator privilege (+o)
/mode #test +o user
/mode #test -o user

# User limit (+l)
/mode #test +l 5
/mode #test -l

# 6. Operator Actions
# Syntax: /kick <channel> <nickname> [reason]
/kick #test user reason

# Syntax: /invite <nickname> <channel>
/invite user #test

# 7. Quit
# Syntax: /quit [reason]
/quit Bye everyone!
